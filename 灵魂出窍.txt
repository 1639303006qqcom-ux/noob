-- 圆球拖动
local ballDragging, ballStartPos, ballStartMouse, ballConn = false, nil, nil, nil

Ball.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        ballDragging = true
        ballStartMouse = input.Position
        ballStartPos = Ball.Position
        if ballConn then ballConn:Disconnect() end
        ballConn = RunService.RenderStepped:Connect(function()
            if ballDragging then
                local m = UIS:GetMouseLocation()
                local d = m - ballStartMouse
                local ss = ScreenGui.AbsoluteSize
                Ball.Position = UDim2.new(0, math.clamp(ballStartPos.X.Offset + d.X, 0, ss.X - 50), 0, math.clamp(ballStartPos.Y.Offset + d.Y, 0, ss.Y - 50))
            end
        end)
    end
end)

UIS.InputEnded:Connect(function(input)
    if ballDragging then
        ballDragging = false
        if ballConn then ballConn:Disconnect() ballConn = nil end
    end
end)

-- 圆球点击切换面板
Ball.MouseButton1Click:Connect(function()
    task.wait(0.05)
    if not ballDragging then
        if Main.Visible then
            Main.Visible = false
            Ball.Text = "▶"
        else
            Main.Visible = true
            Ball.Text = "🎥"
        end
    end
end)

-- ============ 自由相机 ============
local savedCameraCFrame = nil

function ToggleFreeCam(enabled)
    freeCamEnabled = enabled
    if enabled then
        oldCameraCFrame = Camera.CFrame
        oldCameraSubject = Camera.CameraSubject
        if lplr.Character and lplr.Character:FindFirstChild("HumanoidRootPart") then
            Camera.CFrame = lplr.Character.HumanoidRootPart.CFrame + Vector3.new(0, 5, -10)
        end
        Camera.CameraSubject = nil
        Camera.CameraType = Enum.CameraType.Scriptable
        if isMobile and getfenv()._mc then
            getfenv()._mc.leftFrame.Visible = true
            getfenv()._mc.upBtn.Visible = true
            getfenv()._mc.downBtn.Visible = true
        end
        StatusLabel.Text = "运行中"
    else
        Camera.CameraType = Enum.CameraType.Custom
        if oldCameraSubject then Camera.CameraSubject = oldCameraSubject
        elseif lplr.Character and lplr.Character:FindFirstChild("HumanoidRootPart") then
            Camera.CameraSubject = lplr.Character.HumanoidRootPart
        end
        if isMobile and getfenv()._mc then
            getfenv()._mc.leftFrame.Visible = false
            getfenv()._mc.upBtn.Visible = false
            getfenv()._mc.downBtn.Visible = false
        end
        StatusLabel.Text = "已关闭"
    end
end

-- ============ 传送功能 ============
function TeleportToCamera()
    if not freeCamEnabled then
        StatusLabel.Text = "请先启动自由视角"
        return
    end
    if not lplr.Character or not lplr.Character:FindFirstChild("HumanoidRootPart") then
        StatusLabel.Text = "角色未加载"
        return
    end
    savedCameraCFrame = lplr.Character.HumanoidRootPart.CFrame
    lplr.Character.HumanoidRootPart.CFrame = Camera.CFrame + Camera.CFrame.LookVector * 3
    StatusLabel.Text = "已传送到相机位置"
end

function TeleportBack()
    if not savedCameraCFrame then
        StatusLabel.Text = "没有保存的位置"
        return
    end
    if not lplr.Character or not lplr.Character:FindFirstChild("HumanoidRootPart") then return end
    lplr.Character.HumanoidRootPart.CFrame = savedCameraCFrame
    savedCameraCFrame = nil
    StatusLabel.Text = "已返回"
end

-- ============ 相机移动 ============
local mouseDelta = Vector2.zero
UIS.InputChanged:Connect(function(input, gp)
    if gp then return end
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        mouseDelta = mouseDelta + input.Delta
    end
end)

RunService:BindToRenderStep("FreeCamUpdate", Enum.RenderPriority.Camera.Value, function(dt)
    if not freeCamEnabled then return end
    dt = math.min(dt, 0.1)
    local mv = Vector3.zero
    local ld = Vector2.zero

    if isMobile then
        if math.abs(leftStickPos.X) > 0.05 or math.abs(leftStickPos.Y) > 0.05 then
            mv = Camera.CFrame.LookVector * (-leftStickPos.Y) + Camera.CFrame.RightVector * leftStickPos.X
        end
        if upHeld then mv = mv + Vector3.new(0, 1, 0) end
        if downHeld then mv = mv + Vector3.new(0, -1, 0) end
        ld = touchLookDelta * 0.5
        touchLookDelta = Vector2.zero
    else
        if UIS:IsKeyDown(Enum.KeyCode.W) then mv = mv + Camera.CFrame.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.S) then mv = mv - Camera.CFrame.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.A) then mv = mv - Camera.CFrame.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.D) then mv = mv + Camera.CFrame.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.E) then mv = mv + Vector3.new(0, 1, 0) end
        if UIS:IsKeyDown(Enum.KeyCode.Q) then mv = mv + Vector3.new(0, -1, 0) end
        ld = mouseDelta * mouseSensitivity
        mouseDelta = Vector2.zero
    end

    if mv.Magnitude > 0.01 then
        Camera.CFrame = Camera.CFrame + mv.Unit * cameraSpeed * dt
    end
    if ld.Magnitude > 0.01 then
        Camera.CFrame = Camera.CFrame
            * CFrame.Angles(0, -ld.X * 0.01, 0)
            * CFrame.Angles(-ld.Y * 0.01, 0, 0)
    end
end)

UIS.InputBegan:Connect(function(input, gp)
    if gp then return end
    if input.UserInputType == Enum.UserInputType.MouseWheel and freeCamEnabled then
        cameraSpeed = math.max(1, cameraSpeed + (input.Position.Z > 0 and 10 or -10))
        SpeedInput.Text = tostring(cameraSpeed)
        SpeedLabel.Text = "飞行速度: " .. cameraSpeed
        StatusLabel.Text = "速度: " .. cameraSpeed
    end
end)
